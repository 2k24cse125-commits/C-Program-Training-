// Online C compiler to run C program online
#include <stdio.h>

int main() {
    int n=5;
    for(int row=1;row<=n;row++){
        for(int col=1;col<=n;col++){
            if(row==1 || row==n||col==1||col==n)
            printf("* ");
            else{
                printf("  ");
            }
        }
        printf("\n");
    }
    
    return 0;
}

int main() {
    int n=5;
    for(int row=1;row<=n;row++){
         for(int col=n-row;col>=1;col--){
            printf(" ");
        }
        for(int col=1;col<=row;col++){
            printf("* ");
        }
        
        printf("\n");
    }
